1.-Se ingresan tres notas de un alumno, si el promedio es mayor o igual a siete mostrar el mensaje 'Promocionado'. Tener en cuenta que para obtener el promedio debemos operar suma=nota1+nota2+nota3; y luego hacer promedio=suma/3.


2.-Solicitar que se ingrese dos veces una clave. Mostrar un mensaje si son iguales (tener en cuenta que para ver si dos variables tienen el mismo valor almacenado debemos utilizar el operador ==)

        function notas() {
            var nota1 = parseInt(prompt('Ingrese la nota del alumno'));
            var nota2 = parseInt(prompt('Ingrese la nota del alumno'));
            var nota3 = parseInt(prompt('Ingrese la nota del alumno'));
            var suma = (nota1 + nota2 + nota3);
            var promedio = suma/3;

            if (promedio>=7) {
                var decision = 'promocionado';
            } else {
                var decision = 'reprobado';
            } 
            document.write('El alumno esta ' + decision);
        }

        function compararClaves(){
            var clave1 = prompt('Ingrese su clave');
            var clave2 = prompt('Ingrese de nuevo su clave');

            if (clave1==clave2) {
                var confirmar = 'son iguales'
            } else {
                var confirmar = 'son diferentes'
            }
            document.write('Las claves ingresadas ' + confirmar)
        }
